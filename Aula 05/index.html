            import pandas as pd

                def remove_outliers(df, column):
                    # Calcular Q1 (25º percentil) e Q3 (75º percentil)
                    Q1 = df[column].quantile(0.25)
                    Q3 = df[column].quantile(0.75)
                    
                    # Calcular o intervalo interquartílico (IQR)
                    IQR = Q3 - Q1
                    
                    # Definir os limites inferior e superior
                    lower_bound = Q1 - 1.5 * IQR
                    upper_bound = Q3 + 1.5 * IQR
                    
                    # Remover outliers
                    filtered_df = df[(df[column] >= lower_bound) & (df[column] <= upper_bound)]
                    
                    return filtered_df

                # Exemplo de uso
                # Criando um DataFrame de exemplo
                data = {'valor': [10, 12, 14, 15, 16, 100, 18, 19, 21, 22, 23, 25, 30]}
                df = pd.DataFrame(data)

                # Remover outliers da coluna 'valor'
                df_sem_outliers = remove_outliers(df, 'valor')

                print("DataFrame sem outliers:")
                print(df_sem_outliers)
